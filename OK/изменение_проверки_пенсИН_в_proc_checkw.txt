  For Rec In (select ok_personal.TABNO,OK_PERSONAL.Nps_Id from OK_PERSONAL where OK_PERSONAL.STATUS_CODE IN (2,5) )
    Loop
      begin
        if  (Rec.Nps_Id is null or Rec.Nps_Id=0 or length(nvl(Rec.Nps_Id,'0'))<14)
        or (substr(to_char(Rec.Nps_Id),1,1) not in ('3','4'))
        then
          err('Ошибка для отчетности ИИАС, Отдел Кадров:(222) Ошибка в длине номере пенсионного счета Таб.Номер='||rec.TABNO||' Пенсионный счет='||rec.Nps_Id);
        end if;
      exception when others then
        err('Ошибка для отчетности ИИАС, Отдел Кадров:(225) Ошибка в номере пенсионного счета Таб.Номер='||rec.TABNO||' Пенсионный счет='||rec.Nps_Id);
      end;
  End Loop;
